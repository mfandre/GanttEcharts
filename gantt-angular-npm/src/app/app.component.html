Installation

npm install echarts -S
npm install ngx-echarts -S
npm install resize-observer-polyfill -D
npm install iamferraz-gantt-chart -S

Usage

<iamferraz-gantt 
  [chartTitle]="'Gantt by André'" 
  [(taskData)]="taskData" <!-- TaskModel[] -->
  [enableDataZoom]="enableDataZoom" 
  [enableDarkTheme]="enableDarkTheme" 
  (editClicked)="onEditClicked($event)"> 
  [colours]="["#f00","#0f0", "#00f"]"
  [enableGroup]="enableGroup"
  [dateFormat]="{MM}/{dd}/{yyyy}"
  [loading]="false|true"
  [translation]="translation" <!-- object with keys: DONE, TO_END, DELAYED -->
</iamferraz-gantt>
    
translation = {
  DONE: "done",
  TO_END: "days to finish",
  DELAYED: "delayed",
  JANUARY : "Jan",
  FEBRUARY : "Fev",
  MARCH : "Mar",
  APRIL : "Apr",
  MAY : "May",
  JUNE : "Jun",
  JULY : "Jul",
  AUGUST : "Aug",
  SEPTEMBER : "Sep",
  OCTOBER : "Oct",
  NOVEMBER : "Nov",
  DECEMBER : "Dec"
}

Dont forget to import the NgxEchartsModule into your app module!

import { NgxEchartsModule } from 'ngx-echarts';

@NgModule({
  imports: [
    ...,
    NgxEchartsModule.forRoot({
      echarts: () => import('echarts')
      })
  ],
  })
export class AppModule { }
